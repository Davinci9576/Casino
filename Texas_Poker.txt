struct Card{

class Deck{
    private:
    vector <Card> cards;
    public:
    Deck(){
        cardBuild();
    }
    void cardBuild(){
        cards.clear();
        for(int v=1; v<=13; v++){
            for(int s=1; s<=4; s++){
                cards.push_back({v, s});
            }
        }
    }
    void shuffleDeck(){
    static random_device rd;
    static mt19937 g(rd());
    shuffle(cards.begin(), cards.end(), g);
    }
    Card dealCard(){
        Card c=cards.back();
        cards.pop_back();
        return c;
    }
};

class HandEvaluator{
    public:
    struct HandResult{
        int rank;
        vector<int> values;
    };
    static HandResult evaluate(vector <Card> player, vector<Card> community){
        vector <Card> all =player;
        all.insert(all.end(), community.begin(), community.end());
        sort(all.begin(), all.end(), [] (Card a, Card b){
            return a.value>b.value;
        });
    
    map <int, int> freq;
    for(auto &c: all){
        freq[c.value]++;
    }
    bool flush =isFlush(all);
    bool straigth= isStraigth(all);
    int pairs=0, three=0, four=0;
    for(auto&f: freq){
        if(f.second==2) pairs++;
        else if(f.second==3) three++;
        else if(f.second==4) four++;
    }
    HandResult result;
    if (straigth && flush){
        map <int, int> suits;
        int i=-1;
        for(auto &c: all){
            suits[c.suit]++;
            if(suits[c.suit]>=5){
                i=c.suit;
            }
        }
        if(i!=-1){
        vector<int> vals;
        for(auto &c: all){
            if(find(vals.begin(), vals.end(), c.value)==vals.end() && c.suit==i){
                vals.push_back(c.value);
            }
        }
        sort(vals.begin(), vals.end(), [](int a, int b){
            return a>b;
        });
        int countss=1;
        int k=-1;
        
        for(size_t i=0; i+1<vals.size(); i++){
            if(vals[i]==vals[i+1]+1){
                countss++;
                if(countss>=5){
                    k=i;
                    break;
                }
            }else{
                countss=1;
            }
        }
        if(k!=-1){
        result.rank=9;
        for(int i=k-4; i<=k; i++){
            result.values.push_back(vals[i]);
            if(result.values.size()>=5){
                break;
            }
        }
    }
}
    }else if(four){
        result.rank=8;
        int quad;
        for(auto &f: freq){
            if(f.second==4){
                quad=f.first;
                break;
            }
        }
        result.values.push_back(quad);
    }else if(three && pairs){
        result.rank=7;
        int triple;
        for(auto &f: freq){
            if(f.second==3){
                triple=f.first;
                break;
            }
        }
        result.values.push_back(triple);
    }else if(flush){
        result.rank=6;
        map<int, int> suits;
        int i=0;
        for(auto &c: all){
            suits[c.suit]++;
            if(suits[c.suit]>=5){
                i=c.suit;
                break;
            }
        }
        for(auto &c: all){
            if(i==c.suit){
                result.values.push_back(c.value);
                if(result.values.size()>=5){
                    break;
                }
            }
        }
    }else if(straigth){
        result.rank=5;
        int count=1;
        int k=-1;
        vector<int> storage;
        for(auto &c: all){
            if(find(storage.begin(), storage.end(), c.value)==storage.end()){
                storage.push_back(c.value);
            }
        }
        sort(storage.begin(), storage.end(), [](int a, int b){
            return a>b;
        });
        for(size_t i=0; i+1<storage.size(); i++){
            if(storage[i]==storage[i+1]+1){
                count++;
                if(count>=5){
                    k=i;
                    break;
                }
            }else{
                count=1;
            }
        }
        result.values.clear();
        if(k!=-1){
        for(int i=k-4; i<=k; i++){
            result.values.push_back(storage[i]);
        }
    }
    }else if(three){
        result.rank=4;
        result.values.clear();
        int tripleValue;
        for(auto & f: freq){
            if(f.second==3){
                tripleValue=f.first;
            }
        }
        result.values.push_back(tripleValue);
        for(auto & c: all){
            if(c.value!=tripleValue){
                result.values.push_back(c.value);
                if(result.values.size()==3){
                    break;
                }
            }
        }
    }else if(pairs==2){ 
        result.rank=3;
        result.values.clear();
        vector <int> pairValue;
        int i=0;
        for(auto &f: freq){
            if(f.second==2){
                pairValue.push_back(f.first);
            }
        }
        sort(pairValue.begin(), pairValue.end(), [] (int a, int b){
            return a>b;
        });
        for(int i=0; i<2; i++){
            result.values.push_back(pairValue[i]);
        }
        for(auto &c: all){
            if(c.value!=pairValue[0] && c.value!=pairValue[1]){
                result.values.push_back(c.value);
                break;
            }
        }
    }else if(pairs==1){
        result.rank=2;
        result.values.clear();
        int pairValue=0;
        for(auto &f: freq){
            if(f.second==2){
                pairValue=f.first;
            }
        }
        result.values.push_back(pairValue);
        for(auto &c: all){
            if(pairValue!=c.value){
                result.values.push_back(c.value);
            }
            if(result.values.size()==4){
                break;
            }
        }
    }else{
        result.rank=1;
        for(auto& c: all){
            result.values.push_back(c.value);
            if(result.values.size()==5){
                break;
            }
        }
    }
    return result;
}
    private:
    static bool isFlush(vector <Card> cards){
        map <int, int> suitCount;
        for(auto &c: cards){
            suitCount[c.suit]++;
            if(suitCount[c.suit]>=5){
                return true;
            }
        }
        return false;
    }
   
    static bool isStraigth(vector <Card> cards){
        vector<int> vals;
        for(auto &c: cards){
            if(find(vals.begin(), vals.end(), c.value)==vals.end()){
                vals.push_back(c.value);
            }
        }
        sort(vals.begin(), vals.end());
        int count=1;
        for(int i=0; i<vals.size()-1; i++){
            if(vals[i]==vals[i+1]-1){
                count++;
                if(count>=5){
                    return true;
                }
            }else{
                count=1;
            }
        }
        return false;
    }
};
class Game{
    private:
    Deck deck;
    Player &player;
    Player dealer;
    vector<Card> communityCards;
    public:
    int choice;
    Game(Player &p): player(p){}
    void startRound(){
        player.clearHand();
        dealer.clearHand();
        communityCards.clear();
        deck.cardBuild();
        deck.shuffleDeck();
        cout<<"\n----HOLE CARDS----\n";
        dealHoleCards();
        player.showHand();
        player.ResetTotalBet();
        
        if(!handAction()){
            return;
        }else{
            cout<<"====Flop======"<<endl;
            dealFlop();
            showCommunity();
            cout<<"Player cards:"<<endl;
            player.showHand();
        }
        if(!handAction()){
            return;
        }else{
            dealTurn();
            cout<<"\n=====Turn=====\n";
            showCommunity();
            cout<<"Player cards:"<<endl;
            player.showHand();
        }
        if(!handAction()){
            return;
        }else{
            dealRiver();
            cout<<"\n=====River=======\n";
            showCommunity();
            cout<<"Player cards:"<<endl;
            player.showHand();
        }
        if(!handAction()){
            return;
        }
        cout<<"\n=======SHOWDOWN ===========\n";
        showTable();
        auto playerResult=HandEvaluator::evaluate(player.getHand(), communityCards);
        auto dealerResult=HandEvaluator::evaluate(dealer.getHand(), communityCards);
        int result=CompareHands(playerResult, dealerResult);
        player.subtractBalance();
        if(result>0){
            cout<<"Player won!"<<endl;
            player.addBalance();
            player.addWins();
            player.GetWinrate();
        }else if(result<0){
            cout<<"Computer won!"<<endl;
            player.addLosses();
            player.GetWinrate();
        }else{
            cout<<"Perfect tie!"<<endl;
            player.PerfectTie();
        }
    }

        bool handAction(){
        choice=playerAction();
        if(choice==1){
            return true;
        }else if(choice==2){
            player.ResetCurrentBet();
            increaseBet();
            return true;
        }
        else if(choice==3){
            player.gameOver();
            return false;
        }else{
            cout<<"Invalid input!"<<endl;
            return false;
        }
    }
    int CompareHands(const HandEvaluator::HandResult &a,const HandEvaluator::HandResult &b){
        if(a.rank>b.rank){
            return 1;
        }else if(b.rank>a.rank){
            return -1;
        }
        for(size_t i=0; i<a.values.size(); i++){
            if(a.values[i]>b.values[i]){
                return 1;
            }else if(b.values[i]>a.values[i]){
                return -1;
            }
        }
        return 0;
    }

    void dealHoleCards(){
        for(int i=0; i<2; i++){
        player.addCard(deck.dealCard());
        dealer.addCard(deck.dealCard());
        }
    }
    void dealFlop(){
        for(int i=0; i<3; i++){
        communityCards.push_back(deck.dealCard());
        }
    }
    void dealTurn(){
        communityCards.push_back(deck.dealCard());
    }
    void dealRiver(){
        communityCards.push_back(deck.dealCard());
    }
    void showCommunity(){
        cout<<"\nCommunity Cards:\n";
            for(Card cards: communityCards){
                cout<<cards.value<<" "<<cards.suit<<endl;
            }
    }
    void showTable(){
        cout<<"Player cards:"<<endl;
        player.showHand();
        cout<<endl<<"Computer cards:"<<endl;
        dealer.showHand();
        cout<<"Community cards:"<<endl;
        for(Card cards: communityCards){
            cout<<cards.value<<" "<<cards.suit<<endl;
        }
    }
    void increaseBet(){
        double ExtraBet;
        while(true){
            cout<<"Enter the extra bet:"<<endl;
            cin>>ExtraBet;
            if(ExtraBet+50<=player.getBalance()){
                break;
            }
            cout<<"Inssufficient balance please try again!"<<endl;
        }
        player.increaseBet(ExtraBet);
    }
    int playerAction(){
        int choice;
        cout<<"Choose your action:"<<endl;
        cout<<"1. Check in"<<endl;
        cout<<"2.Bet"<<endl;
        cout<<"3.Fold"<<endl;
        cout<<"Choice: ";
        cin>>choice;
        return choice;
    }
};
